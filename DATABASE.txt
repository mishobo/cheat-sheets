# mysql setup 
> docker run -d -p 3306:3306 --name mysqldb -v mysql-data:/var/lib/mysql -e MYSQL_ROOT_PASSWORD=31875595 mysql

# postgres setup
> docker run --name postgresdb -e POSTGRES_PASSWORD=31875595 -d postgres
> docker container exec -it postgres bash
> pg_isready -U postgres
